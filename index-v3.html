                title: 'Ethical AI & Bias Mitigation',

                concept: 'Designing AI to be fair, accountable, and transparent.',

                description: 'This pattern focuses on the critical design considerations to ensure AI systems are developed and used responsibly, addressing issues like fairness, privacy, accountability, and the reduction of harmful biases in generated content.',

                examples: [

                    'Fairness dashboards: Tools that allow developers to monitor and mitigate biases in AI model outputs.',

                    'Bias detection tools: Automated systems that flag potentially biased language or imagery generated by AI.',

                    'Privacy-preserving AI: Techniques like federated learning or differential privacy used to protect user data while training AI models.',

                    'Transparency reports: Public documentation detailing an AI system\'s ethical guidelines, data sources, and known limitations regarding bias.'

                ],

                icon: `<svg class="pattern-card-icon icon-pink" xmlns="http://www.w3.org/2000/svg" width="24" height="24" viewBox="0 0 24 24" fill="none" stroke="currentColor" stroke-width="2" stroke-linecap="round" stroke-linejoin="round"><path d="M12 22s8-4 8-10V5l-8-3-8 3v7c0 6 8 10 8 10Z"/><path d="m9 12 2 2 4-4"/></svg>`

            }

        ];


        // Get DOM elements

        const patternsGrid = document.getElementById('patternsGrid');

        const modalOverlay = document.getElementById('patternModalOverlay');

        const modalCloseButton = document.getElementById('modalCloseButton');

        const modalIconWrapper = document.getElementById('modalIconWrapper');

        const modalTitle = document.getElementById('modalTitle');

        const modalDescription = document.getElementById('modalDescription');

        const modalExamples = document.getElementById('modalExamples');


        /**

         * Creates an HTML element for a pattern card.

         * @param {Object} pattern - The pattern data.

         * @returns {HTMLElement} The created pattern card element.

         */

        function createPatternCard(pattern) {

            const card = document.createElement('div');

            card.className = 'pattern-card';

            card.setAttribute('data-pattern-id', pattern.id); // Store ID for easy lookup


            card.innerHTML = `

                <div class="pattern-card-icon-wrapper">

                    ${pattern.icon}

                </div>

                <h2>${pattern.title}</h2>

                <p>${pattern.concept}</p>

            `;


            // Add click event listener to open the modal

            card.addEventListener('click', () => openModal(pattern));

            return card;

        }


        /**

         * Opens the modal and populates it with the given pattern data.

         * @param {Object} pattern - The pattern data to display in the modal.

         */

        function openModal(pattern) {

            modalIconWrapper.innerHTML = pattern.icon; // Set the icon

            // Adjust icon color for modal, if needed (re-apply class for larger context)

            const iconSvg = modalIconWrapper.querySelector('svg');

            if (iconSvg) {

                 // Remove existing icon color classes (e.g., icon-indigo)

                iconSvg.className.baseVal = iconSvg.className.baseVal.replace(/icon-\w+/g, '');

                // Add the correct color class to the SVG in the modal context

                const colorClass = `icon-${pattern.icon.match(/icon-(\w+)"/)?.[1] || 'indigo'}`; // Extract color from initial icon SVG string

                iconSvg.classList.add(colorClass);

            }



            modalTitle.textContent = pattern.title;

            modalDescription.textContent = pattern.description;


            // Clear previous examples

            modalExamples.innerHTML = '';

            // Add new examples

            pattern.examples.forEach(example => {

                const li = document.createElement('li');

                li.textContent = example;

                modalExamples.appendChild(li);

            });


            // Show the modal

            modalOverlay.classList.add('active');

            document.body.style.overflow = 'hidden'; // Prevent scrolling background

        }


        /**

         * Closes the modal.

         */

        function closeModal() {

            modalOverlay.classList.remove('active');

            document.body.style.overflow = ''; // Restore scrolling

        }


        // Add event listener to the modal close button

        modalCloseButton.addEventListener('click', closeModal);


        // Add event listener to close modal when clicking outside the content (on the overlay)

        modalOverlay.addEventListener('click', (event) => {

            if (event.target === modalOverlay) {

                closeModal();

            }

        });


        // Initialize: Render all pattern cards

        patterns.forEach(pattern => {

            const card = createPatternCard(pattern);

            patternsGrid.appendChild(card);

        });
