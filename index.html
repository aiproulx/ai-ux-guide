    Chosen Palette: Calm Tech (Warm Neutrals)

    Application Structure Plan: A multi-section, single-page application designed for exploration. It starts with a foundational overview, moving to core principles. The centerpiece is an interactive "Pattern Explorer" with filtering capabilities, allowing users to dive into specific categories and patterns. Subsequent sections cover domain-specific applications, ethical challenges (visualized with a chart), and future trends. This thematic, non-linear structure was chosen over a simple linear report to encourage user discovery, facilitate comparison, and make the dense information highly digestible and engaging, catering to designers and developers looking for specific information.

    Visualization & Content Choices:

    - Traditional vs. AI UX: Goal: Compare. Method: Two-column interactive diagram (HTML/CSS). Interaction: Hover to reveal details. Justification: More dynamic than a static table.

    - Core Principles: Goal: Inform & Relate. Method: Clickable cards. Interaction: Click to expand and reveal description. Justification: Links abstract principles to concrete patterns.

    - AI UX Pattern Explorer: Goal: Organize & Explore. Method: Filterable grid of cards. Interaction: Click filters to see categories; click cards to see details in a modal. Justification: Manages a large amount of information effectively, making it the app's core tool.

    - Ethical Challenges: Goal: Relate. Method: Horizontal Bar Chart. Interaction: Hover tooltips provide context. Justification: Visually connects problems (challenges) to solutions (design patterns) in a quantifiable way. Library: Chart.js.

    - Future Trends: Goal: Inform & Inspire. Method: Infographic-style cards. Interaction: Subtle hover animations. Justification: Makes forward-looking content visually appealing.

    CONFIRMATION: NO SVG graphics used. NO Mermaid JS used.

            const patternFilters = document.getElementById('pattern-filters');

            const patternsGrid = document.getElementById('patterns-grid');

            const domainTabs = document.getElementById('domain-tabs');

            const domainContent = document.getElementById('domain-content');

            const futureTrendsGrid = document.getElementById('future-trends-grid');

            

            const modal = document.getElementById('pattern-modal');

            const modalBg = modal.querySelector('.modal-bg');

            const modalClose = document.getElementById('modal-close');

            const modalTitle = document.getElementById('modal-title');

            const modalCategory = document.getElementById('modal-category');

            const modalBody = document.getElementById('modal-body');


            // Render Principles

            appData.principles.forEach(p => {

                const div = document.createElement('div');

                div.className = 'bg-white p-6 rounded-xl shadow-md border border-stone-200 text-center';

                div.innerHTML = `

                    <div class="text-4xl">${p.icon}</div>

                    <h3 class="mt-4 text-xl font-semibold text-stone-900">${p.title}</h3>

                    <p class="mt-2 text-stone-600">${p.description}</p>

                `;

                principlesGrid.appendChild(div);

            });


            // Render Pattern Filters and Grid

            const categories = ['all', ...new Set(appData.patterns.map(p => p.category))];

            categories.forEach(cat => {

                const button = document.createElement('button');

                button.dataset.filter = cat;

                button.className = 'filter-btn px-4 py-2 text-sm font-semibold rounded-full transition-colors ' + (cat === 'all' ? 'bg-teal-600 text-white' : 'bg-white text-stone-600 hover:bg-teal-100');

                button.textContent = cat.charAt(0).toUpperCase() + cat.slice(1);

                patternFilters.appendChild(button);

            });

            

            const renderPatterns = (filter = 'all') => {

                patternsGrid.innerHTML = '';

                const filteredPatterns = filter === 'all' ? appData.patterns : appData.patterns.filter(p => p.category === filter);

                filteredPatterns.forEach(p => {

                    const div = document.createElement('div');

                    div.className = 'pattern-card bg-white p-6 rounded-xl shadow-lg border border-stone-200 cursor-pointer';

                    div.dataset.patternId = p.id;

                    div.dataset.category = p.category;

                    div.innerHTML = `

                        <h4 class="font-bold text-lg text-stone-900">${p.title}</h4>

                        <p class="mt-2 text-sm text-stone-600">${p.definition}</p>

                        <span class="mt-4 inline-block text-xs font-semibold text-teal-600 uppercase tracking-wider">${p.category}</span>

                    `;

                    patternsGrid.appendChild(div);

                });

            };


            patternFilters.addEventListener('click', e => {

                if (e.target.classList.contains('filter-btn')) {

                    const filter = e.target.dataset.filter;

                    document.querySelectorAll('.filter-btn').forEach(btn => {

                        btn.classList.remove('bg-teal-600', 'text-white');

                        btn.classList.add('bg-white', 'text-stone-600', 'hover:bg-teal-100');

                    });

                    e.target.classList.add('bg-teal-600', 'text-white');

                    e.target.classList.remove('bg-white', 'text-stone-600', 'hover:bg-teal-100');

                    renderPatterns(filter);

                }

            });


            patternsGrid.addEventListener('click', e => {

                const card = e.target.closest('.pattern-card');

                if (card) {

                    const patternId = card.dataset.patternId;

                    const pattern = appData.patterns.find(p => p.id === patternId);

                    if (pattern) {

                        modalTitle.textContent = pattern.title;

                        modalCategory.textContent = pattern.category;

                        modalBody.innerHTML = `

                            <p class="text-stone-700">${pattern.definition}</p>

                            <h5 class="font-semibold text-stone-800 mt-4 mb-1">Why It's Important</h5>

                            <p class="text-stone-700">${pattern.importance}</p>

                            <h5 class="font-semibold text-stone-800 mt-4 mb-1">Example</h5>

                            <p class="text-stone-700 bg-stone-100 p-3 rounded-md">${pattern.example}</p>

                        `;

                        modal.classList.remove('hidden');

                        modal.classList.add('flex');

                        setTimeout(() => {

                            modal.querySelector('.modal-content').classList.remove('opacity-0', 'scale-95');

                            modal.querySelector('.modal-bg').classList.remove('opacity-0');

                        }, 10);

                    }

                }

            });


            const closeModal = () => {

                modal.querySelector('.modal-content').classList.add('opacity-0', 'scale-95');

                modal.querySelector('.modal-bg').classList.add('opacity-0');

                setTimeout(() => {

                    modal.classList.add('hidden');

                    modal.classList.remove('flex');

                }, 300);

            }

            modalClose.addEventListener('click', closeModal);

            modalBg.addEventListener('click', closeModal);

            

            renderPatterns();


            // Render Domain Tabs & Content

            const renderDomainContent = (domainId) => {

                const domain = appData.domains.find(d => d.id === domainId);

                if (!domain) return;

                

                domainContent.innerHTML = `

                    <div class="max-w-4xl mx-auto">

                        <p class="text-center text-stone-600">${domain.description}</p>

                        <div class="mt-6 space-y-4">

                            ${domain.patterns.map(p => `

                                <div class="bg-white p-4 rounded-lg border border-stone-200">

                                    <h4 class="font-semibold text-stone-800">${p.title}</h4>

                                    <p class="text-sm text-stone-600 mt-1">${p.description}</p>

                                </div>

                            `).join('')}

                        </div>

                    </div>

                `;


                document.querySelectorAll('.domain-tab').forEach(tab => {

                    tab.classList.remove('text-teal-600', 'border-teal-600');

                    tab.classList.add('text-stone-500', 'border-transparent', 'hover:text-stone-700', 'hover:border-stone-300');

                });

                const activeTab = document.querySelector(`.domain-tab[data-domain='${domainId}']`);

                if (activeTab) {

                    activeTab.classList.add('text-teal-600', 'border-teal-600');

                    activeTab.classList.remove('text-stone-500', 'border-transparent', 'hover:text-stone-700', 'hover:border-stone-300');

                }

            };

            

            appData.domains.forEach(domain => {

                const button = document.createElement('button');

                button.className = 'domain-tab py-4 px-1 border-b-2 font-medium text-sm';

                button.textContent = domain.title;

                button.dataset.domain = domain.id;

                domainTabs.appendChild(button);

            });


            domainTabs.addEventListener('click', e => {

                if (e.target.classList.contains('domain-tab')) {

                    renderDomainContent(e.target.dataset.domain);

                }

            });

            renderDomainContent(appData.domains[0].id);


            // Render Future Trends

            appData.futureTrends.forEach(trend => {

                const div = document.createElement('div');

                div.className = 'text-center p-6 bg-white rounded-xl shadow-md border border-stone-200';

                div.innerHTML = `

                    <div class="text-5xl">${trend.icon}</div>

                    <h3 class="mt-5 text-xl font-semibold text-stone-900">${trend.title}</h3>

                    <p class="mt-2 text-stone-600">${trend.description}</p>

                `;

                futureTrendsGrid.appendChild(div);

            });


            // Ethics Chart

            const ctx = document.getElementById('ethicsChart').getContext('2d');

            const ethicsData = {

                labels: ['AI Bias', 'Lack of Transparency', 'User Privacy', 'Uncertainty Mgmt', '"Uncanny Valley"'],

                datasets: [{

                    label: 'Related Design Solutions',

                    data: [4, 4, 4, 3, 3], 

                    backgroundColor: [

                        'rgba(204, 102, 102, 0.6)',

                        'rgba(102, 153, 204, 0.6)',

                        'rgba(102, 204, 153, 0.6)',

                        'rgba(255, 159, 64, 0.6)',

                        'rgba(153, 102, 255, 0.6)'

                    ],

                    borderColor: [

                        'rgba(204, 102, 102, 1)',

                        'rgba(102, 153, 204, 1)',

                        'rgba(102, 153, 204, 1)',

                        'rgba(255, 159, 64, 1)',

                        'rgba(153, 102, 255, 1)'

                    ],

                    borderWidth: 1

                }]

            };


            new Chart(ctx, {

                type: 'bar',

                data: ethicsData,

                options: {

                    indexAxis: 'y',

                    responsive: true,

                    maintainAspectRatio: false,

                    plugins: {

                        legend: {

                            display: false

                        },

                        title: {

                            display: true,

                            text: 'Number of Key Design Solutions per Ethical Challenge'

                        },

                        tooltip: {

                            callbacks: {

                                label: function(context) {

                                    let label = context.dataset.label || '';

                                    if (label) {

                                        label += ': ';

                                    }

                                    if (context.parsed.x !== null) {

                                        label += context.parsed.x + ' patterns';

                                    }

                                    return label;

                                }

                            }

                        }

                    },

                    scales: {

                        x: {

                            beginAtZero: true,

                            title: {

                                display: true,

                                text: 'Number of Solution Patterns'

                            }

                        }

                    }

                }

            });


            // Mobile menu toggle

            const mobileMenuButton = document.getElementById('mobile-menu-button');

            const mobileMenu = document.getElementById('mobile-menu');

            mobileMenuButton.addEventListener('click', () => {

                mobileMenu.classList.toggle('hidden');

            });

            mobileMenu.addEventListener('click', () => {

                mobileMenu.classList.add('hidden');

            });



            // Nav link active state on scroll

            const sections = document.querySelectorAll('section');

            const navLinks = document.querySelectorAll('.nav-link');


            const observer = new IntersectionObserver((entries) => {

                entries.forEach(entry => {

                    if (entry.isIntersecting) {

                        navLinks.forEach(link => {

                            link.classList.remove('active');

                            if (link.getAttribute('href').substring(1) === entry.target.id) {

                                link.classList.add('active');

                            }

                        });

                    }

                });

            }, { rootMargin: '-50% 0px -50% 0px' });


            sections.forEach(section => {

                observer.observe(section);

            });

        });

    </script>
